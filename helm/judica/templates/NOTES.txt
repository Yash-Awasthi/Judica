1. Get the application URL by running these commands:
{{- if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts }}
  {{- range .paths }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}{{ .path }}
  {{- end }}
{{- end }}
{{- else if contains "NodePort" .Values.service.frontend.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "aibyai.fullname" . }}-frontend)
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.frontend.type }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch the status of by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "aibyai.fullname" . }}-frontend'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "aibyai.fullname" . }}-frontend --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.frontend.port }}
{{- else if contains "ClusterIP" .Values.service.frontend.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "{{ include "aibyai.frontend.selectorLabels" . }}" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace {{ .Release.Namespace }} $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  echo "Visit http://127.0.0.1:4173 to use your application"
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 4173:$CONTAINER_PORT
{{- end }}

2. Backend API is available at:
  kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ include "aibyai.fullname" . }}-backend 3000:{{ .Values.backend.port }}

3. Run database migrations (if needed):
  kubectl --namespace {{ .Release.Namespace }} run aibyai-migrate \
    --image={{ .Values.image.repository }}/backend:{{ .Values.image.tag }} \
    --restart=Never \
    --env="DATABASE_URL=<your-db-url>" \
    -- npx drizzle-kit push
